  <script>
    // Chuyển text trạng thái thành badge (chọn cột 6)
    document.querySelectorAll("table tbody tr").forEach((row) => {
      const statusTd = row.querySelector("td:nth-child(6)");
      if (!statusTd) return;
      const text = statusTd.textContent.trim();

      const span = document.createElement("span");
      span.classList.add("status");

      if (text === "Đang bán") {
        span.classList.add("status-dangban");
        span.textContent = text;
      } else if (text === "Hết hàng") {
        span.classList.add("status-hethang");
        span.textContent = text;
      } else if (text === "Ngưng bán") {
        span.classList.add("status-ngungban");
        span.textContent = text;
      } else {
        span.textContent = text;
      }

      statusTd.innerHTML = "";
      statusTd.classList.add("status-cell");
      statusTd.appendChild(span);
    });

    // Gắn sự kiện cho nút Sửa / Xóa
    function attachActionEvents() {
      document.querySelectorAll(".btn-edit").forEach((btn) => {
        btn.removeEventListener && btn.removeEventListener("click", () => {}); // noop
        btn.addEventListener("click", () => {
          // ví dụ: chuyển tới edit-product.html
          window.location.href = "edit-product.html";
        });
      });

      document.querySelectorAll(".btn-delete").forEach((btn) => {
        btn.addEventListener("click", () => {
          if (confirm("Bạn có chắc muốn xóa sản phẩm này không?")) {
            // ở đây bạn có thể gọi API xóa, hoặc remove hàng:
            const tr = btn.closest("tr");
            if (tr) tr.remove();
            alert("Sản phẩm đã được xóa!");
          }
        });
      });
    }

    // gọi khi DOM sẵn sàng
    attachActionEvents();
  </script>
